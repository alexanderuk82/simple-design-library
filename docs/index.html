03 / Code

Copy. Paste.
Ship.

Import what you need. Customize with props or CSS variables. That's it. No complex setup, no configuration nightmares.

App.tsx 
import { Button, Input, Card } from 'simple-design-system'

export default function App() {
  return (
    <Card>
      <Card.Header>
        <h2>Sign Up</h2>
      </Card.Header>

      <Card.Body>
        <Input
          label="Email"
          type="email"
          placeholder="you@example.com"
        />
        <Button variant="primary" fullWidth>
          Continue
        </Button>
      </Card.Body>
    </Card>
  )
}
